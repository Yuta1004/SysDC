### logger.def

```text
unit logger;

from string import String;
from io import IO;
from time import Time, Timestamp;

module Logger {
    proc info(msg: String) {
        @affect IO.stdout(msg)

        @modify msg {
            use timestamp;
        }

        @spawn timestamp: Timestamp {
            let now = Time.get_now_time();
            return now;
        }
    }

    proc error(msg: String) {
        @affect IO.stdout(msg)

        @modify msg {
            use timestamp;
        }

        @spawn timestamp: Timestamp {
            let now = Time.get_now_time();
            return now;
        }
    }
}
```
